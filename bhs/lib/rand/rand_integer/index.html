Function rand_integer

Description

Generate a random integer number from a certain range.

Signature

rand_integer (

  1. Number Maximum range value [100].
  2. Number Minimum range value [1].

)

Return value

Operation status.

Examples

#!/bin/bash
. ${SHELL_BOOTSTRAP}

lib rand

echo 'Range [-20 - 20]:'
for i in {1..10}; do
   echo $(rand_integer 20 -20)
done

echo 'Tiny range [-1 - 1]:'
for i in {1..10}; do
   echo $(rand_integer 1 -1)
done

echo 'Only maximum value:'
for i in {1..10}; do
   echo $(rand_integer 1000)
done
$ /path/to/foo.sh
