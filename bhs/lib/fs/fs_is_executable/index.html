Function fs_is_executable

Description

Is the file executable?

Signature

fs_is_executable (

  1. String Path to the file.

)

Return value

Operation status.

Examples

#!/bin/bash
. ${SHELL_BOOTSTRAP}

lib fs

fpath_not_executable=/foo/bar/doe
fpath_executable=/tmp

fs_is_executable ${fpath_executable} \
     && printf 'Path "%s" is executable\n' ${fpath_executable} \
     || printf 'Path "%s" is not executable\n' ${fpath_executable} 

fs_is_executable ${fpath_not_executable} \
     && printf 'Path "%s" is executable\n' ${fpath_not_executable} \
     || printf 'Path "%s" is not executable\n' ${fpath_not_executable} 
$ /path/to/foo.sh
Path "/tmp" is executable
Path "/foo/bar/doe" is not executable
