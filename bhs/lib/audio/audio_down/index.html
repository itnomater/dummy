Function audio_down

Description

Decrease the audio volume level.

Signature

audio_down (

  1. Number Percentage of audio volume level to decrease.

)

Return value

none

Examples

#!/bin/bash
. ${SHELL_BOOTSTRAP}

lib audio
 
audio_set 50
printf 'Current volume level is: %s\n' $(audio_get)
audio_down
printf 'Current volume level is: %s\n' $(audio_get)
audio_down 5
printf 'Current volume level is: %s\n' $(audio_get)
$ /path/to/foo.sh
Current volume level is: 50
Current volume level is: 49
Current volume level is: 44
